        For at se om vores hypotese er gældende laves der en Bertlett-test, hvor følgende gælder:
		\begin{eqnarray}\nonumber
		Ba & = & \frac{-2 \ln Q(x)}{C}\\\nonumber
		-2 \ln Q(x) & = & f_{1} \ln s_{1}^{2}- \sum\limits_{i = 1}^{k} f_{(i)} \ln s_{(i)}^{2}\\ \nonumber
		C & = & 1 + \frac{1}{3(k-1)}\Big[\Big(\sum\limits_{i=1}^{k} \frac{1}{f_{(i)}}\Big) - \frac{1}{f_{1}} \Big]\\ \nonumber
		s_{1}^{2} & = & \frac{\sum\limits_{i = 1}^{k} f_{(i)}s_{(i)}^{2}}{\sum\limits_{i = 1}^{k} f_{(i)}} = \frac{\sum\limits_{i = 1}^{k} SSD_{(i)}}{\sum\limits_{i = 1}^{k} f_{(i)}} = \frac{SSD_{1}}{f_{1}}\\ \nonumber
		f_{1} & = & \sum\limits_{i = 1}^{k} f_{(i)} = n. - k\\\nonumber
		p_{obs}(x) & = & 1 - F_{\chi^{2}(k-1)}(Ba)
		\end{eqnarray}
		Bemærk, at ovenstående $p_{obs}(x)$ kun gælder, hvis: $f_{(i)} \geq 2, i = 1, \dots, k$.\\
